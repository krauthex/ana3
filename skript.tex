% user defined TikZ commands
\newcommand{\manifold}[3]{
	\shade[shading=axis,bottom color=gray!90,top color=white, shading angle=35]
	(#1,#2) to[out=45,in=120,looseness=.8] (3*#3+#1,-.4*#3+#2) to[out=45,in=120,looseness=.8] (5*#3+#1,#2) to[out=110,in=70,looseness=1.1] cycle;
} % usage: \manifold{x}{y}{scale}; manifold in tikz, position relative to lower left corner
\newcommand{\hcsphere}[3]{
	\draw[dashed,color=gray!140] (#1,#2) to[out=-90,in=-90,looseness=1.7] (2*#3+#1,#2);
	\shade[shading=axis,bottom color=gray!90,top color=white, shading angle=35] (#1,#2) to[out=90,in=90,looseness=1.7] (2*#3+#1,#2) to[out=-140,in=-60,looseness=.6] cycle;
	\draw (#1,#2) to[out=90,in=90,looseness=1.7] (2*#3+#1,#2) to[out=-140,in=-60,looseness=.6] cycle;
	\draw[dashed,color=gray!140] (2*#3+#1,#2) to[out=115,in=50,looseness=.5] (#1,#2);
} % usage: \hcsphere{x}{y}{scale}; half colored sphere in tikz, position relative to middle left
\newcommand{\saucer}[3]{
	\manifold{#1}{#2}{#3};
	\hcsphere{#1+1.9*#3}{#2+#3}{.7*#3};
} % usage: \saucer{x}{y}{scale}; „flying saucer“ (manifold cut by sphere) in tikz, position relative to lower left corner
\newcommand{\squigglyset}[4]{
	\draw plot[smooth cycle,tension=1] coordinates {
		(#1,#2+#4*1.4) (#1+#3*.3,#2+#4*1.7) (#1+#3*.8,#2+#4*2.2)
		(#1+#3*1.5,#2+#4*1.7) (#1+#3*1.5,#2+#4) (#1+#3*1.6,#2+#4*.4)
		(#1+#3*1.4,#2) (#1+#3*0.8,#2+#4*.2) (#1+#3*.4,#2)
		(#1,#2+#4*.4) (#1+#3*.1,#2+#4*.9)};
} % usage: \squigglyset{x}{y}{scale x}{scale y}; position relative to lower left
